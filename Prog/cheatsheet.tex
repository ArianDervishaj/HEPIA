\begin{enumerate}
    \item char, unsigned char         8-bit
    \item short, unsigned short,      16-bit
    \item int unsigned int,           32-bit
    \item long, unsigned long         64-bit
    \item float                       simple précision
    \item double                      double précision

\end{enumerate}
